#--------------------------- Range Test -----------------------------#
add_executable(range_test
    duck_range_test.c
)
target_link_libraries(range_test 
    PRIVATE
        lora
        gps
        board_control
        hardware_spi 
        hardware_uart 
        hardware_gpio
)
# create UF2 files
pico_add_extra_outputs(range_test)
# Enable stdout to USB serial
pico_enable_stdio_usb(range_test 1)
# Disable stdout to UART0
pico_enable_stdio_uart(range_test 0)
